## üíæ Database Implementation

### SQL Tables

```sql
-- Sections table
CREATE TABLE crypto_sections (
    section_id INTEGER PRIMARY KEY,
    section_name VARCHAR(100),
    description TEXT,
    total_categories INTEGER,
    total_subtopics INTEGER
);

-- Categories table
CREATE TABLE crypto_categories (
    category_id VARCHAR(10) PRIMARY KEY,
    section_id INTEGER REFERENCES crypto_sections,
    category_name VARCHAR(200),
    description TEXT,
    total_subtopics INTEGER
);

-- Subtopics table
CREATE TABLE crypto_subtopics (
    subtopic_id VARCHAR(20) PRIMARY KEY,
    category_id VARCHAR(10) REFERENCES crypto_categories,
    subtopic_name VARCHAR(300),
    description TEXT,
    priority VARCHAR(20),
    target_qa_pairs INTEGER DEFAULT 30,
    actual_qa_pairs INTEGER DEFAULT 0
);

-- Q&A pairs table
CREATE TABLE crypto_qa_pairs (
    qa_id SERIAL PRIMARY KEY,
    subtopic_id VARCHAR(20) REFERENCES crypto_subtopics,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    difficulty_level VARCHAR(20),
    tags TEXT[],
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
