CREATE TABLE fundamentals_qa_pairs (
    qa_id INTEGER PRIMARY KEY AUTOINCREMENT,
    category_id INTEGER,
    subtopic_id INTEGER,
    pair_number INTEGER,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    answer_length INTEGER,
    batch_number INTEGER,

    category_name TEXT NOT NULL,
    category_description TEXT,

    subtopic_id INTEGER PRIMARY KEY AUTOINCREMENT,
    category_id INTEGER,
    subtopic_name TEXT NOT NULL,
    subtopic_description TEXT,
    target_pairs INTEGER DEFAULT 100,
    completed_pairs INTEGER DEFAULT 0,
    FOREIGN KEY (category_id) REFERENCES fundamentals_categories(category_id)
);
