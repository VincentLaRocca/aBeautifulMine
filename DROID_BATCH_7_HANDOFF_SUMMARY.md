import json

from pathlib import Path

def import_batch_7():
    """Import all Batch 7 JSON files into database"""

    db = sqlite3.connect('crypto_indicators.db')
    cursor = db.cursor()

    inbox = Path('inbox/droid')
    json_files = list(inbox.glob('*_qa_pairs.json'))

    total_imported = 0

    for json_file in json_files:
        with open(json_file, 'r', encoding='utf-8') as f:
            data = json.load(f)

        indicator_name = json_file.stem.replace('_qa_pairs', '')

        for qa_pair in data.get('questions', []):
            cursor.execute('''
                INSERT INTO crypto_indicators_qa
                (indicator_name, question, answer, indicator_category, session_number)
                VALUES (?, ?, ?, ?, ?)
            ''', (
                indicator_name,
                qa_pair.get('question'),
                qa_pair.get('answer'),
                determine_category(indicator_name),
                determine_session(indicator_name)
            ))
            total_imported += 1

    db.commit()
    print(f"‚úÖ Imported {total_imported} Q&A pairs from {len(json_files)} indicators")
